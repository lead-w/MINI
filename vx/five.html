# 2、点击播放和暂停按钮，实现布局的动画

  • transition 过度动画, 需要有状态的变化
    • transition-property 规定设置过渡效果的 CSS 属性的名称。
    • transition-duration 规定完成过渡效果需要多少秒或毫秒。
    • transition-timing-function 规定速度效果的速度曲线。
      • ease 默认
      • linear 线性
    • transition-delay 定义过渡效果何时开始。
  • animation 关键帧动画，与元素变化无关，相对于多个补间动画
    • animation-name 规定需要绑定到选择器的 keyframe 名称
    • animation-duration 规定完成动画所花费的时间，以秒或毫秒计
    • animation-timing-function 规定动画的速度曲线
      • ease 默认 慢快慢
      • linear 匀速
    • animation-delay 规定在动画开始之前的延迟
    • animation-iteration-count 规定动画应该播放的次数
      • infinite 无限次
      • n 几次
    • animation-direction 规定是否应该轮流反向播放动画
  • @keyframes 创建动画
    • 0-100%
    • from（与 0% 相同）
    • to（与 100% 相同）

# 4、获取歌曲的播放路径 创建背景音乐

  • wx.getBackgroundAudioManager 创建音乐对象的实例
    • src 设置音乐播放地址
    • title 设置播放的歌曲
    • play 播放音乐
    • pause 暂停播放
async onLoad(options) {
    const songId = options.songId;
    this.setData({
      songId,
    })
    // 存储了歌曲id之后 我们想让歌曲播放
    // 1、根据id获取歌曲的详情
    await  this.getSongDetail()
    // 2、还要获取一下播放歌曲的路径
    await this.getSongUrl()
    // 3、播放歌曲
    this.setData({
      isPlay: true
    })
    if(!this.player){
      // 判断有没有音频播放管理器对象，如果没有直接创建，如果没有直接创建修改src和title
      this.player = wx.getBackgroundAudioManager()
    }

    this.player.src = this.data.songUrl;
    this.player.title = this.data.songDetail;
    console.log('options',options)
}

# 5、处理暂停和播放

handleMusicPlay(){
    let isPlay = this.data.isPlay;
    this.setData({
      isPlay: !isPlay
    })
    this.playOrPauseMusic()
},

// 点击播放 或者 暂停歌曲
playOrPauseMusic(){
    let isPlay = this.data.isPlay
    isPlay?this.player.play():this.player.pause();
}

# 6、系统内置的播放

  • 这里有系统内置的播放状态， 我们的按钮播放要与他同步
  • 利用 播放器的监听事件 onPlay 和 onPause
  • 通过上面创建音乐实例 调用onPlay和onPause监控音乐的播放和暂停,在里面写同步一些按钮信息
async onLoad(options) {
    const songId = options.songId;
    this.setData({
      songId,
    })
    // 存储了歌曲id之后 我们想让歌曲播放
    // 1、根据id获取歌曲的详情
    await  this.getSongDetail()
    // 2、还要获取一下播放歌曲的路径
    await this.getSongUrl()
    // 3、播放歌曲
    this.setData({ 
      isPlay: true
    })
    if(!this.player){
      // 判断有没有音频播放管理器对象，如果没有直接创建，如果没有直接创建修改src和title
      this.player = wx.getBackgroundAudioManager()
    }
    this.player.src = this.data.songUrl;
    this.player.title = this.data.songDetail.name;

    // 这里有系统内置的播放状态，  我们的按钮播放要与他同步
    this.player.onPlay(()=>{
      this.setData({
        isPlay:true
      })
    })
    this.player.onPause(()=>{
      this.setData({
        isPlay:false 
      })
    })
  }

# 8、上一曲下一曲逻辑， 消息的订阅和发布

// recommend.js
// 订阅详情页 点击上一曲或者下一曲 
PubSub.subscribe('songType',(msg,type)=>{
    // 根据当前保存的歌曲id 找到当前这首歌
    let index = this.data.recommendList.findIndex(item => item.id ===     Number(this.data.songId));
    let newIndex = 0;
    if(type === 'next'){
    newIndex = index === this.data.recommendList.length - 1 ? 0 : index + 1
    }else{
    newIndex = index === 0 ? this.data.recommendList.length - 1 : index - 1
    }
    PubSub.publish('newSongId', this.data.recommendList[newIndex].id)
    this.setData({
    songId:this.data.recommendList[newIndex].id
    })
})

// detail.js
// 订阅
PubSub.subscribe('newSongId',async (msg,id)=>{
    this.setData({
        songId:id,
    })
    // 存储了歌曲id之后 我们想让歌曲播放
    // 1、根据id获取歌曲的详情
    await  this.getSongDetail()
    // 2、还要获取一下播放歌曲的路径
    await this.getSongUrl()
    // 3、播放歌曲
    this.setData({
    isPlay: true
    })
    this.player = wx.getBackgroundAudioManager()
    this.player.src = this.data.songUrl;
    this.player.title = this.data.songDetail.name;
})
// 点击上一首和下一首
handleSwitch(event){
    let type = event.currentTarget.id;
    console.log(type)
    PubSub.publish('songType',type)
}

# 10、进度条布局

  • 1、引入moment 库 npm i moment
  • 2、绑定监听歌曲播放进度
    • onTimeUpdate 监听
  • currentTime 获取当前的时间
  • duration 获取总时间
  • 进度条布局的实现:有三层
    • 1、第一层:最外成一个打低色的进度条(白色)
    • 2、第二层:里面第一层和父元素的高度一致,宽度则是根据 当前播放的时间点动态计算
      • 父总宽*currentTime/duration + 'rpx'
    • 3、第三层:小圆点,通过绝对定位,始终在父元素的最右边
<view class="progress">
    <text class="currentTime">{{startTime}}</text>
    <view class="progressBar">
      <view class="timeBar" style="width:{{width}}">
        <view class="circle"></view>
      </view>
    </view>
    <text class="totaltTime">{{totaltTime}}</text>
</view>
<script>
  // 绑定监听歌曲播放进度
  this.player.onTimeUpdate(()=>{
    // console.log(this.player.currentTime)
    // console.log(this.player.duration)
    const startTime = moment(this.player.currentTime * 1000).format('mm:ss')
    const totalTime = moment(this.player.duration * 1000).format('mm:ss')
    const width = 450 * this.player.currentTime / this.player.duration + 'rpx'
    this.setData({
      startTime,
      totalTime,
      width
    })
  })
</script>

# 常规分包

  • 分为主包和分包
  • 1、主包一般都要有启动页/tabBar页面
  • 2、分包可以使用主包的资源,反之则不行
  • 3、分包的小程序目录结构如下：
├── app.js
├── app.json
├── app.wxss
├── packageA
│   └── pages
│       ├── cat
│       └── dog

│   └── pages
│       ├── apple
│       └── banana
├── pages
│   ├── index

# 独立分包

  • 添加一个字段"independent": true就变为独立分包
    • 独立分包可以不用加载主包
  • 当依赖主包的资源的时候出问题
    • 需要给分包拷贝一份资源
    • 需要在分包内部也引入一次
    • 如果一些页面功能比较独立，可以使用独立分包

{
  "pages": [
    "pages/index/index",
    "pages/video/video",
    "pages/center/center",
    "pages/login/login"
  ],
  "subpackages": [
    {
      "root": "packageA",
      "pages": [
        "pages/recommend/recommend",
        "pages/detail/detail"
      ],
      "independent": true
    }
  ]
}

# 分包预下载

  • 在进入小程序某个页面时，由框架自动预下载可能需要的分包，提升进入后续分包页面时的启动速度。对于独立分包，也可以预下载主包。
  • 配置方法: 预下载分包行为在进入某个页面时触发，通过在 app.json 增加 preloadRule 配置来控制。
    • packages 进入页面后预下载分包的 root 或 name。__APP__表示主包。
    • network 在指定网络下预下载，可选值为：
      • all: 不限网络
      • wifi: 仅 wifi 下预下载
    • preloadSubpackages: success 访问对应的页面,则表示成功
{
  "pages": ["pages/index"],
  "subpackages": [
    {
      "root": "important",
      "pages": ["index"],
    },
    {
      "root": "sub1",
      "pages": ["index"],
    },
    {
      "name": "hello",
      "root": "path/to",
      "pages": ["index"]
    },
    {
      "root": "sub3",
      "pages": ["index"]
    },
    {
      "root": "indep",
      "pages": ["index"],
      "independent": true
    }
  ],
  "preloadRule": {
    "pages/index": {
      "network": "all",
      "packages": ["important"]
    },
    "sub1/index": {
      "packages": ["hello", "sub3"]
    },
    "sub3/index": {
      "packages": ["path/to"]
    },
    "indep/index": {
      "packages": ["__APP__"]
    }
  }
}
