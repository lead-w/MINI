import request from '../../utils/request.js'

const state = {
	categoryList:[],
};

const mutations = {
	SET_CATEGORYLIST(state,categoryList){
			state.categoryList = categoryList
	},
};

const actions = {
	async getCategoryList({commit}){
		const result = await request('/getCategoryData')
		commit('SET_CATEGORYLIST',result)
	}
};

const getters = {
	 navList(state){
		 return (state.homeData.kingKongModule || {}).kingKongList || []
	 },
};

export default {
		state,
		mutations,
		actions,
		getters
}
  • 分页页面的静态数据

    • 2、左边tab 点击的时候 会动态获取右边列表的数据,我们可以动态修改defaultIndex的数据
    • 3、本身右边的数据就是根据列表的总数据,获取其中一条,我在点击左边tab的时候 动态修改defaultIndex既可以

			</div>

				<div class="scrollItem">

<script>
	import { mapState } from 'vuex'
 export default {
  data() {
   return {
		defaultIndex:0
   }
  },
  methods: {
   getCategoryList(){
		 this.$store.dispatch('getCategoryList')
	 }
  },

			categoryList: state => state.categorys.categoryList
		}),

	},

 }

# 添加购物车的操作

  • uni.switchTab 跳转到 tabBar 页面，并关闭其他所有非 tabBar 页面。
  • 在首页 => 居家生活 => 列表中的商品 => 滑动到最下面 点击加入购物车
  • 1、判断购物车列表中，有没有当前添加的这个商品详情页数据
  • 2、有就添加数据 没有就push一个数据
    • isChecked修改为true count+1
  • 3、没有则就添加一组数据
    • isChecked修改为true count = 1, 通过dispatch 新增这个对象
  • 注意$set用法
// store/module/shopcart
const mutations = {
	ADD_SHOPCART(state,shopCart){
		state.shopCartList.push(shopCart)
	}
};
// detail.vue
computed:{
	...mapState({
		shopCartList: state => state.shopCart.shopCartList
	})
},
methods:{
	addShopCart(){
		// 判断购物车列表中，有没有当前添加的这个商品详情页数据
		// 有就添加数据 没有就push一个数据
		let obj = this.shopCartList.find(item => item.id === this.goodsInfo.id)
		if(obj){
			// 代表购物车当中已存在， 我只需要改变数量和状态
			obj.isChecked = true
			obj.count += 1
		}else{
			obj = this.goodsInfo
			// 数据能添加 但是后期添加的时候不能响应式
			obj['isChecked'] = true;
			obj['count'] = 1
			// 数组响应式
			// this.$set(obj,'isChecked',true)
			// this.$set(obj,'count',1)
			this.$store.commit('ADD_SHOPCART', obj);
		}
	}
}
